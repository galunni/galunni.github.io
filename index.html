Funzioni: 

  size(larghezza, altezza); // dimensioni finestra
  background(rgb);          // colore di sfondo espresso in un numero da 0 (nero) a 255 (bianco)

  point(x, y);           // punto nella posizione x y
  line(x1, y1, x2, y2);  // linea che parte da x1 y1 ed arriva ad x2 y2
  ellipse(x, y, larghezza, altezza);   // ellisse con centro nella posizione x y
  rect(x, y, larghezza, altezza);      // rettangolo nella posizione x y (punto in alto a sinistra)

  strokeWeight(larghezza);  // dimensioni delle linee, punti e bordi espressa in pixel
  noStroke();               // non usare i bordi

  fill(180);                // colore usato per riempore le forme espresso in un numero da 0 (nero) a 255 (bianco)
