git init create a repository in current directory cd your_directory; git init
git clone clone a repository in actual directory git clone /path/to/directory # clone a repository from local filesystem
git clone username@host:/path/to/repository # clone a repository from remote
git checkout — . Jump back to the last Commit and remove any untracked changes (if not added yet)
git add adds files/directories to the index(stage) so they will be included by the commit command git add my_file.txt
git add . # using wildcards
git add *
git text_??.txt
git reset unstage all files in index (not committed yet, but added) git reset file.txt # unstage a specific file
git commit commit all files that are on stage (added to the index with git add) git commit -m “Commit Message, useful comments“
git log List all Commits inside selected Branch git log --author=Linus # Filter by author
git log --pretty=oneline # compressed log
git status show current status of repository (untracked, added and modified files)
git diff Show changes between commits, commit and working tree, etc
git branch list all branches in repository and show selected branch git branch -D new-branch # delete the new-branch branch
git checkout switch to a branch git checkout master # switch to master branch
git checkout -b new-stuffs # create a new branch called new-stuffs and select it
git merge branch-name Merge branch-name in the currently selected branch
git rm remove from git git rm file1.txt # set file for remove (in stage) and delete from file system
git commit -m "remove file1.txt" # also remove file in repository
rm -rf .git stop using git on the current project
git fetch Downloads all history from the remote tracking branches
git merge Combines remote tracking branch into current local branch
git push Uploads all local branch commits to remote repository
git pull Updates your current local working branch with all new commits from the corresponding remote branch on GitHub. git pull is a combination of git fetch and git merge
